    // Function to fetch page content with proxy fallback

    async function fetchWithProxy(url) {

        for (let i = 0; i < PROXIES.length; i++) {

            try {

                const proxyUrl = PROXIES[i] + encodeURIComponent(url);

                const response = await fetch(proxyUrl, {

                    headers: {

                        'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36'

                    }

                });

                if (response.ok) {

                    return await response.text();

                }

            } catch (e) {

                console.log(`Proxy ${i} failed:`, e);

                continue;

            }

        }

        throw new Error('All proxies failed');

    }

    

    // Extract image URLs from Pinterest page HTML

    function extractImagesFromHTML(html, boardUrl) {

        const imageUrls = new Set();

        

        // Method 1: Find all img tags with pinimg.com

        const imgRegex = /<img[^>]+src=["'](https?:\/\/i\.pinimg\.com\/[^"']+\.(jpg|png|jpeg|webp))["']/gi;

        let match;

        while ((match = imgRegex.exec(html)) !== null) {

            let imgUrl = match[1];

            // Convert to original size

            imgUrl = imgUrl.replace(/\/\d+x\d+/, '/originals');

            imgUrl = imgUrl.replace(/\/\d+x\//, '/originals/');

            imageUrls.add(imgUrl);

        }

        

        // Method 2: Find images in JSON data (Pinterest stores rich data)

        const jsonRegex = /<script[^>]*type=["']application\/json["'][^>]*>([\s\S]*?)<\/script>/gi;

        let jsonMatch;

        while ((jsonMatch = jsonRegex.exec(html)) !== null) {

            try {

                const data = JSON.parse(jsonMatch[1]);

                const extractFromObj = (obj) => {

                    if (!obj) return;

                    if (typeof obj === 'string') {

                        if (obj.includes('i.pinimg.com') && (obj.includes('.jpg') || obj.includes('.png') || obj.includes('.jpeg'))) {

                            let clean = obj.replace(/\/\d+x\d+/, '/originals');

                            imageUrls.add(clean);

                        }

                    } else if (Array.isArray(obj)) {

                        obj.forEach(extractFromObj);

                    } else if (typeof obj === 'object') {

                        Object.values(obj).forEach(extractFromObj);

                    }

                };

                extractFromObj(data);

            } catch(e) {}

        }

        

        // Method 3: Look for data-src and other attributes

        const dataSrcRegex = /data-src=["'](https?:\/\/i\.pinimg\.com\/[^"']+\.(jpg|png|jpeg|webp))["']/gi;

        while ((match = dataSrcRegex.exec(html)) !== null) {

            let imgUrl = match[1].replace(/\/\d+x\d+/, '/originals');

            imageUrls.add(imgUrl);

        }

        

        return Array.from(imageUrls);

    }

    

    // Download and create ZIP file

    async function downloadImagesAsZip(imageUrls, boardName) {

        const JSZip = window.JSZip;

        if (!JSZip) {

            throw new Error('JSZip library not loaded');

        }

        

        const zip = new JSZip();

        let downloaded = 0;

        let failed = 0;

        

        updateStatus(`📸 downloading ${imageUrls.length} images...`, true, false, `0/${imageUrls.length}`);

        

        // Download images in parallel with concurrency limit

        const concurrency = 5;

        const batches = [];

        for (let i = 0; i < imageUrls.length; i += concurrency) {

            batches.push(imageUrls.slice(i, i + concurrency));

        }

        

        for (let batchIndex = 0; batchIndex < batches.length; batchIndex++) {

            const batch = batches[batchIndex];

            const promises = batch.map(async (imgUrl, idx) => {

                try {

                    const response = await fetch(imgUrl, {

                        headers: { 'User-Agent': 'Mozilla/5.0' }

                    });

                    if (response.ok) {

                        const blob = await response.blob();

                        const ext = imgUrl.includes('.png') ? 'png' : 'jpg';

                        const filename = `image_${String(downloaded + failed + idx + 1).padStart(3, '0')}.${ext}`;

                        zip.file(filename, blob);

                        downloaded++;

                    } else {

                        failed++;

                    }

                } catch (e) {

                    failed++;

                    console.warn(`Failed: ${imgUrl}`);

                }

                updateStatus(`💖 downloading images...`, true, false, `${downloaded + failed}/${imageUrls.length}`);

            });

            await Promise.all(promises);

        }

        

        updateStatus(`✨ zipping ${downloaded} images...`, true, false);

        

        const zipBlob = await zip.generateAsync({ type: 'blob' });

        return { zipBlob, downloaded, failed };

    }

    

    // Main function to process Pinterest URL

    async function processPinterestUrl(pinterestUrl) {

        // Clean and validate URL

        let cleanUrl = pinterestUrl.trim();

        if (!cleanUrl.startsWith('http')) {

            cleanUrl = 'https://' + cleanUrl;

        }

        

        if (!cleanUrl.includes('pinterest.com') && !cleanUrl.includes('pin.it')) {

            throw new Error('Please enter a valid Pinterest URL 💕');

        }

        

        // Expand pin.it shortlinks

        if (cleanUrl.includes('pin.it')) {

            updateStatus(`🔗 expanding Pinterest link...`, true, false);

            const response = await fetch(cleanUrl, { method: 'HEAD', redirect: 'follow' });

            cleanUrl = response.url;

        }

        

        updateStatus(`🌸 loading Pinterest board...`, true, false);

        

        // Fetch the page

        const html = await fetchWithProxy(cleanUrl);

        

        if (!html || html.length < 500) {

            throw new Error('Could not load the Pinterest page. Make sure the board is public!');

        }

        

        updateStatus(`🔍 finding all beautiful images...`, true, false);

        

        // Extract images

        const imageUrls = extractImagesFromHTML(html, cleanUrl);

        

        if (imageUrls.length === 0) {

            throw new Error('No images found 😢. Is this board public? Try another board or pin!');

        }

        

        // Remove duplicates and limit to reasonable amount

        const uniqueImages = [...new Set(imageUrls)];

        const maxImages = 200; // Safe limit for browser performance

        const imagesToDownload = uniqueImages.slice(0, maxImages);

        

        updateStatus(`🎀 found ${uniqueImages.length} images! ${uniqueImages.length > maxImages ? `(downloading first ${maxImages})` : ''}`, true, false);

        

        // Generate board name for ZIP

        let boardName = 'pinterest_board';

        const boardMatch = cleanUrl.match(/pinterest\.com\/[^\/]+\/([^\/?]+)/);

        if (boardMatch) {

            boardName = boardMatch[1];

        }

        

        // Download and create ZIP

        const { zipBlob, downloaded, failed } = await downloadImagesAsZip(imagesToDownload, boardName);

        

        if (downloaded === 0) {

            throw new Error('Could not download any images. Please try again!');

        }

        

        // Trigger download

        const downloadUrl = URL.createObjectURL(zipBlob);

        const a = document.createElement('a');

        a.href = downloadUrl;

        a.download = `${boardName}_pinterest_images.zip`;

        document.body.appendChild(a);

        a.click();

        document.body.removeChild(a);

        URL.revokeObjectURL(downloadUrl);

        

        updateStatus(`🎉 SUCCESS! Downloaded ${downloaded} images ${failed > 0 ? `(${failed} failed)` : ''} 🎉`, false, false);

        return true;

    }

    // Load JSZip library dynamically

    function loadJSZip() {

        return new Promise((resolve, reject) => {

            if (window.JSZip) {

                resolve();

                return;

            }

            const script = document.createElement('script');

            script.src = 'https://cdnjs.cloudflare.com/ajax/libs/jszip/3.10.1/jszip.min.js';

            script.onload = resolve;

            script.onerror = reject;

            document.head.appendChild(script);

        });

    }

    

    // Initialize

    loadJSZip().catch(() => {
